### 1. 코드

<br>

```cpp
#include <iostream>
#include <algorithm>
#include <queue>
#include <string.h>
#define MAX 9

using namespace std;

//row, column
int R, C, zeroCnt, MAXzero;
int dR[] = { 1,0,-1,0 };
int dC[] = { 0,1,0,-1 };

int map[MAX][MAX], tmp[MAX][MAX];
queue <pair <int, int>> baseQ;

bool isInRange(int row, int col) {
	return (row >= 0 && row < R) && (col >= 0 && col < C);
}

//바이러스가 퍼진 이후의 0의 개수를 셈
void BFS(int & zero) {
	
	int afterWall[MAX][MAX];
	memcpy(afterWall, tmp, sizeof(tmp));

	int zC = zero; //0의 개수
	queue <pair<int, int>> Q = baseQ;

	//bfs 알고리즘

	while (!Q.empty()) {
		
		int cR = Q.front().first;
		int cC = Q.front().second;

		Q.pop();

		for (int i = 0; i < 4; i++) {
			
			int temp_cR = cR + dR[i];
			int temp_cC = cC + dC[i];

			if (isInRange(temp_cR, temp_cC) && !afterWall[temp_cR][temp_cC]) {
				afterWall[temp_cR][temp_cC] = 2;
				Q.push({ temp_cR, temp_cC });
				zC--; //0의 개수 줄임
			}
		}
	}

	MAXzero = max(MAXzero, zC);
}

void makeWall(int cnt, int & zero) {
	//zero는 0의 개수
	//cnt는 새로 세운 벽의 개수 (최대 3개)

	if (cnt == 3) {
		BFS(zero);
		return;
	}

	for (int i = 0; i < R; i++) {
		for (int j = 0; j < C; j++) {
			if (tmp[i][j] == 0) {
				tmp[i][j] = 1;
				zero--;

				makeWall(cnt + 1, zero);

				tmp[i][j] = 0;
				zero++;
			}
		}
	}
}

int main() {
	cin >> R >> C;

	zeroCnt = R * C; // 0의 개수
	for (int i = 0; i < R; i++) {
		for (int j = 0; j < C; j++) {
			cin >> map[i][j];
			if (map[i][j] != 0) {
				--zeroCnt;
				
				//바이러스 위치 큐에 저장
				if (map[i][j] == 2)
					baseQ.push({ i,j });
			}
		}
	}

	for (int i = 0; i < R; i++) {
		for (int j = 0; j < C; j++) {
			//0 발견시 연구실 상태를 복사
			if (map[i][j] == 0) {
				memcpy(tmp, map, sizeof(map));
				tmp[i][j] = 1; //벽을 하나 세움
				--zeroCnt;

				makeWall(1, zeroCnt);

				tmp[i][j] = 0; //벽을 없앰
				zeroCnt++;
			}
		}
	}

	cout << MAXzero << endl;

	return 0;
}
```

<br>

### 2. 코멘트

  BFS로 병균이 퍼져나가는 것을 구현했다. 
  벽을 세우는 알고리즘은 main()함수 내에서 처음에 임의로 벽 하나 세우고,
  makeWall()이라는 함수에서 재귀 돌면서 벽 두개를 세웠다. 
  이런식으로 하면 주어진 맵에서 3개의 벽을 세우는 모든 경우의 수를 구할 수 있다.
