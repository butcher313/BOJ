### 1. 코드
```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

#define MAX 1000
using namespace std;

int N, M;
int map[MAX][MAX];
vector <pair <int, int>> tomato;
vector <int> num;
int row[] = { 1, 0, -1, 0 };
int col[] = { 0, 1, 0, -1 };

bool is_in_range(int r, int c) {
	return (0 <= r && r < N) && (0 <= c && c < M);
}

void bfs(pair <int, int> point) {
	
	queue <pair<int, int>> q;
	q.push(point);

	while (!q.empty()) {
		int r = q.front().first;
		int c = q.front().second;

		q.pop();
		for (int i = 0; i < 4; i++) {
			int temp_r = r + row[i];
			int temp_c = c + col[i];

			if (!is_in_range(temp_r, temp_c) || map[temp_r][temp_c] == -1) continue;
			
			map[temp_r][temp_c]++;
			q.push(make_pair(temp_r, temp_c));

			/*
			//범위 밖, 처음부터 익어있는 토마토, 토마토가 없는경우, 숫자를 차감하려고 보니 본인보다 수가 낮은경우
			if (!is_in_range(temp_r, temp_c) || map[temp_r][temp_c] == 1 || 
				map[temp_r][temp_c] == -1 || map[temp_r][temp_c] <= map[r][c]) continue;
			
			if(map[temp_r][temp_c] > 1)
				num.push_back(map[temp_r][temp_c]--);
			else
				num.push_back(map[temp_r][temp_c]++);
				
			q.push(make_pair(temp_r, temp_c));
			num.push_back(map[temp_r][temp_c]);
			*/
		}
	}
}

int main() {
	cin >> M >> N;

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			cin >> map[i][j];
			if (map[i][j] == 1)
				tomato.push_back(make_pair(i, j));
		}
	}

	for(auto t : tomato)
		bfs(t);

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			cout << map[i][j];
		}
		cout << endl;
	}

}
```
